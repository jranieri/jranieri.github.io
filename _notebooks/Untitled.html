class points():
    
    def __init__(self, N):
        self.N=N
        self.points=Set([])
        
    def addPoints(self,x,y):
        self.points.add([x,y])
        
    def compute_distances(self):
        self.distances=Set([])
        for i in self.points:
            for j in self.points:
                self.distances.add(abs(i[0]-i[1]))
